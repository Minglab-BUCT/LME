13pub fn auto_connect_bonds(
14    atoms: &Vec<Atom3D>,
15    r_cov_table: &RadiisTable,
16) -> Result<Vec<(usize, usize, f64)>> {
17    let mut bonds = vec![];
18    for (a_idx, atom) in atoms.iter().enumerate() {
19        let r_a = r_cov_table
20            .get(atom.element)
21            .with_context(|| {
22                format!(
23                    "Failed to found the radiis for the second atom element {}",
24                    atom.element
25                )
26            })?
27            .value;
28        let p_a = atom.position;
29        for (b_idx, atom) in atoms.iter().enumerate().skip(a_idx + 1) {
30            let r_b = r_cov_table
31                .get(atom.element)
32                .with_context(|| {
33                    format!(
34                        "Failed to found the radiis for the second atom element {}",
35                        atom.element
36                    )
37                })?
38                .value;
39            let distance = (atom.position - p_a).norm();
40            if distance <= r_a + r_b {
41                bonds.push((a_idx, b_idx, 1.0))
42            }
43        }
44    }
45    Ok(bonds)
46}

48pub fn molecular_graph_walk(
49    graph: &StableUnGraph<Atom3D, f64, usize>,
50    entry: usize,
51    current_depth: usize,
52    limit_depth: usize,
53    excludes: Vec<usize>,
54) -> Result<Vec<(usize, Atom3D)>> {
55    let current_position = graph
56        .node_weight(entry.into())
57        .with_context(|| format!("Failed to get atom information with index {}", entry))?;
58    let neighbors = graph
59        .neighbors(entry.into())
60        .filter(|neighbor| !excludes.contains(&neighbor.index()))
61        .collect::<Vec<_>>();
62    if current_depth == limit_depth || neighbors.len() == 0 {
63        Ok(vec![(entry, *current_position)])
64    } else {
65        let sub_find_results = neighbors
66            .into_iter()
67            .map(|index| {
68                molecular_graph_walk(
69                    graph,
70                    index.index(),
71                    current_depth + 1,
72                    limit_depth,
73                    vec![vec![entry.index()], excludes.clone()].concat(),
74                )
75            })
76            .collect::<Result<Vec<_>>>()?;
77        Ok(sub_find_results.concat())
78    }
79}

83pub fn get_molecular_graph(
84    atoms: &Vec<Atom3D>,
85    bonds: &Vec<(usize, usize, f64)>,
86) -> MolecularGraph {
87    let mut molecular_graph: StableUnGraph<Atom3D, f64, usize> = StableUnGraph::default();
88    for atom in atoms {
89        molecular_graph.add_node(*atom);
90    }
91    molecular_graph.extend_with_edges(bonds);
92    molecular_graph
93}

97pub fn sterimol(molecular_graph: &MolecularGraph, table: &RadiisTable) -> Result<(f64, f64, f64)> {
98    let a = molecular_graph
99        .node_weight(0.into())
100        .with_context(|| "First atom of substituent group not found, require at least 2 atoms")?;
101    let b = molecular_graph
102        .node_weight(1.into())
103        .with_context(|| "Second atom of subsitutent group not found, require at least 2 atoms")?;
104    let b_radii = table
105        .get(b.element)
106        .with_context(|| format!("Unable to get radii from table for element {}", b.element))?
107        .value;
108    let ab = b.position - a.position;
109    let axis = ab.normalize();
110    let l = molecular_graph
111        .node_indices()
112        .skip(2)
113        .map(|idx| molecular_graph.node_weight(idx).unwrap())
114        .map(|atom| {
115            let radii = table
116                .get(atom.element)
117                .with_context(|| format!("Failed to read radiis of element {}", atom.element))?
118                .value;
119            let projection = (atom.position - a.position).dot(&axis);
120            Ok::<f64, anyhow::Error>(projection + radii)
121        })
122        .collect::<Result<Vec<f64>>>()?
123        .into_iter()
124        .reduce(|acc, next| if acc > next { acc } else { next })
125        .unwrap_or(ab.norm() + b_radii);
126    let branches = molecular_graph_walk(&molecular_graph, 1, 0, 1, vec![0])?
127        .into_iter()
128        .map(|(idx, _)| {
129            Ok(
130                molecular_graph_walk(&molecular_graph, idx, 1, 0, vec![0, 1])?
131                    .into_iter()
132                    .map(|(_, atom)| atom)
133                    .map(|atom| {
134                        let c_radii = table
135                            .get(atom.element)
136                            .with_context(|| {
137                                format!("Failed to read radiis of element {}", atom.element)
138                            })?
139                            .value;
140                        let bc = atom.position - b.position;
141                        let projection = bc.dot(&axis) * axis;
142                        let distance = (bc - projection).norm();
143                        Ok(distance + c_radii)
144                    })
145                    .collect::<Result<Vec<_>>>()?
146                    .into_iter()
147                    .reduce(|acc, next| if acc > next { acc } else { next })
148                    .expect("At least one value in each branch here"),
149            )
150        })
151        .collect::<Result<Vec<_>>>()?;
152    let b1 = branches
153        .iter()
154        .copied()
155        .reduce(|acc, next| if acc < next { acc } else { next })
156        .unwrap_or(b_radii);
157    let b5 = branches
158        .into_iter()
159        .reduce(|acc, next| if acc > next { acc } else { next })
160        .unwrap_or(b_radii);
161    Ok((l, b1, b5))
162}

164pub fn tolman_cone_angle(molecular_graph: &MolecularGraph) -> Result<f64> {
165    let a = molecular_graph
166        .node_weight(0.into())
167        .with_context(|| "First atom of substituent group not found, require at least 2 atoms")?;
168    let b = molecular_graph
169        .node_weight(1.into())
170        .with_context(|| "Second atom of subsitutent group not found, require at least 2 atoms")?;
171    let ab = b.position - a.position;
172    let branches = molecular_graph
173        .neighbors(1.into())
174        .filter(|idx| idx.index() != 0);
175    let branch_angles = branches
176        .map(|idx| {
177            let branch_atoms =
178                molecular_graph_walk(molecular_graph, idx.index(), 1, 0, vec![0, 1])?;
179            let branch_angle = branch_atoms
180                .into_iter()
181                .map(|(_, atom)| {
182                    let bc = atom.position - b.position;
183                    let theata_main = (bc.dot(&ab) / (bc.norm() * ab.norm())).acos();
184                    Ok(theata_main)
185                })
186                .collect::<Result<Vec<_>>>()?
187                .into_iter()
188                .reduce(|acc, next| if acc > next { acc } else { next })
189                .unwrap_or_default();
190            Ok(branch_angle)
191        })
192        .collect::<Result<Vec<_>>>()?;
193    let branches = branch_angles.len();
194    let tolman_angle = branch_angles
195        .into_iter()
196        .map(|value| value / 2.)
197        .sum::<f64>();
198    if branches == 0 {
199        Ok(0.)
200    } else {
201        Ok(tolman_angle / (branches as f64) * 2.)
202    }
203}
