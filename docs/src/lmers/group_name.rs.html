118#[derive(Deserialize, Serialize)]
119#[serde(untagged)]
120pub enum IndexCollect {
121    Collect(BTreeSet<usize>),
122    Range(RangeInclusive<usize>),
123    Complex {
124        includes: Vec<IndexCollect>,
125        #[serde(default)]
126        excludes: Vec<IndexCollect>,
127    },
128}
129
130impl IndexCollect {
131    fn collect(self) -> BTreeSet<usize> {
132        match self {
133            IndexCollect::Collect(value) => value,
134            IndexCollect::Range(range) => range.collect(),
135            IndexCollect::Complex { includes, excludes } => {
136                let mut content = BTreeSet::new();
137                for include in includes {
138                    content.extend(include.collect());
139                }
140
141                for exclude in excludes {
142                    let exclude = exclude.collect();
143                    content.retain(|index| !exclude.contains(index));
144                }
145                content
146            }
147        }
148    }
149}

153pub enum FriendlyGroupName {

155    Friendly(BTreeMap<String, IndexCollect>),

158impl From<FriendlyGroupName> for GroupName {

160        Self::from_iter(match value {
161            FriendlyGroupName::Friendly(value) => Self::from_iter(
162                value
163                    .into_iter()
164                    .map(|(k, v)| v.collect().into_iter().map(move |v| ((&k).to_string(), v)))
165                    .flatten(),
166            ),
167            FriendlyGroupName::UnFriendly(value) => Self(value),
